Closures: Use Case

var _handleServerResponse(next) {
    return function(err, response) {
        var isError = !!err;
        console.log("Server responded - endpoint: %s isError: %s, response: %o",
            response.endpoint, isError, response);

        if(isError) { console.error(err.stack); }
        var formattedResponse = {
            success: !isError,
            error: err,
            response: response,
            data: (response && response.data && response.data.length >= 0) ? response.data : []
        };

        next(formattedResponse);
    }
}

Publish & Subscribe

/* Notify widgets that model has changed */
EmployeesModel = {
    topicTitle: "EmployeesModel",
    cache: [{name: "Fred", dept: "Sales"}, {name:"Jane", dept: "IT"}],
    addEmployee: function(newEmployee) {
        this.cache.push(newEmployee);
        topic.publish(this.topicTitle, this.cache);

/* Listens for changes to EmployeesModel */
function SalesEmployeesWidget() {
    topic.subscribe(EmployeesModel.topicTitle,
        function(employees) {
            var salesEmployees = employees.filter(isSales);
            this.draw(salesEmployees);
        }.bind(this)
    ),
    this.draw = function(employees) { /* Draw widget */ };
}

function isSales(emp) { return emp.dept == "Sales" };

Deferreds & Promises

/* dojo/Deferred - for Node, see Q: https://github.com/kriskowal/q */
function getEmployeesWithManager(manager) {
    var deferred = new Deferred();
    xhr("/employees", {
        onSuccess: function(data) {
            var managerEmployees =
                data.filter(function(emp) {
                    return emp == manager;
                });
            deferred.resolve(managerEmployees);
        }, onError: function(err) {
            var reason = err.msg;
            deferred.reject(reason);
        }
    );
    return deferred.promise;
}
getEmployeesWithManager("Jane").then(
    function success(janesEmployees) {
        /* Do Something */
    }, function err(errMsg) {
        console.error(errMsg);
    }
);

Aspects

//Called after any EmployeesModel.getManager() call.
aspect.after(EmployeesModel, "getManager", function(manager) {
    //arguments passed to callback are return value of original call
    var returnValue = manager || "No Manager";
    //Return value is what will be returned to the original caller
    return returnValue;
});
//Called before any SalesEmployeesWidget.set() call
aspect.before(SalesEmployeesWidget, "set", function(field, value) {
    //arguments are the original arguments passed to SalesEmployeesWidget.set()
    console.log("Field %s value changed to value %s", field, value);
    //return value is the arguments passed to original function (or next aspect)
    return [field, value];
});
//Replaces original function with advising function
aspect.around(SalesEmployeesWidget, "set", function(originalSet) {
    return function(field, value) {
        //do something before call
        var returnValue = originalSet(field, value);
        //do something after call
        return returnValue; //returned to original caller
    }
});
